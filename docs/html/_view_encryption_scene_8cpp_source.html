239 void ViewEncryptionScene::OnDecryptButtonPressed()
240 {
241  if (m_password.empty())
242  {
243  (void)DialogUtils::Warn("Password is empty");
244  return;
245  }
246 
247 
248  try
249  {
250  // Get encryption cipher from database, we havent got it at the beginning to save up memory
251  auto cipher = Database::GetCipherByEncryptionID(m_encryption->ide);
252  ENIGMA_ASSERT_OR_THROW(cipher, "Failed to get cipher by encryption id");
253 
254  // Auto detect algorithm used for encryption
255  Algorithm::Type algo_type = Algorithm::DetectFromCipher(cipher->data);
256  ENIGMA_INFO("Successfully auto-detected algorithm used for encryption which is {0}", Algorithm::AlgoTypeEnumToStr(algo_type));
257 
258 
259  // Create encryptor based on selected algorithm type
260  const auto algorithm = Algorithm::CreateFromType(algo_type, Algorithm::Intent::Decrypt);
261  ENIGMA_ASSERT_OR_THROW(algorithm, "Failed to create algorithm from type");
262 
263 
264  // Decrypt either file or text ...
265  // ... case file
266  if (m_encryption->is_file)
267  {
268  // Decrypt cipher
269  String buffer = algorithm->Decrypt(m_password, cipher->data);
270  ENIGMA_ASSERT_OR_THROW(!buffer.empty(), "Failed to decrypt file");
271 
272  // Decompress buffer
273  /*
274  Note: You should compress before encrypting. Encryption turns your data into high-entropy data,
275  usually indistinguishable from a random stream. Compression relies on patterns in order to gain
276  any size reduction. Since encryption destroys such patterns, the compression algorithm would be
277  unable to give you much (if any) reduction in size if you apply it to encrypted data.
278  */
279  buffer = GZip::Decompress(buffer);
280  ENIGMA_ASSERT_OR_THROW(!buffer.empty(), "Failed to decompress file buffer");
281 
282  // Ask where to save decrypted file ?
283  // Get path to where decrypted file should be saved
284  if (const String output_filename = SaveFileDialog{"Save Decrypted File To"}.Show(); !output_filename.empty())
285  {
286  // Write file recovered content to output file
287  const bool saved = FileUtils::Write(output_filename, buffer);
288  ENIGMA_ASSERT_OR_THROW(saved, "Failed to save decrypted file");
289 
290  // little happy msg
291  DialogUtils::Info("Successfully decrypted file to " + output_filename);
292  }
293 
294  }
295  // ... case text
296  else
297  {
298  // Decrypt cipher text
299  m_recovered_text = algorithm->Decrypt(m_password, cipher->data);
300  ENIGMA_ASSERT_OR_THROW(!m_recovered_text.empty(), "Failed to decrypt text");
301 
302  // Decompress recovered text
303  /*
304  Note: You should compress before encrypting. Encryption turns your data into high-entropy data,
305  usually indistinguishable from a random stream. Compression relies on patterns in order to gain
306  any size reduction. Since encryption destroys such patterns, the compression algorithm would be
307  unable to give you much (if any) reduction in size if you apply it to encrypted data.
308  */
309  //ENIGMA_LOG("m_recovered_text size before {0}", m_recovered_text.size());
310  m_recovered_text = GZip::Decompress(m_recovered_text);
311  //ENIGMA_LOG("m_recovered_text size after {0}", m_recovered_text.size());
312  ENIGMA_ASSERT_OR_THROW(!m_recovered_text.empty(), "Failed to decompress recovered text");
313  }
314 
315  }
316  catch (const CryptoPP::Exception& e)

318  const String err_msg = CryptoPPUtils::GetFullErrorMessage(e);
319  ENIGMA_ERROR("Decryption Failure: {0}", err_msg);
320  (void)DialogUtils::Error("Decryption Failure", err_msg);
321  }
322  catch (const std::exception& e)

324  ENIGMA_ERROR("Decryption Failure: {0}", e.what());
325  (void)DialogUtils::Error("Decryption Failure", e.what());
326  }
327  catch (...)
328  {
329  const String err_msg = "Decryption Failure: Unknown Error";

331  (void)DialogUtils::Error(err_msg);
