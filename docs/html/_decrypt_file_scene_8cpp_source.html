355 {
356  // Validate fields
357  if (m_in_filename.empty())
358  {
359  (void)DialogUtils::Warn(("Input File to decrypt is empty"));
360  }
361  else if (m_out_filename.empty())

366  {

373  else // Alles gut
374  {
375  try
376  {
377  // Auto detect encryption algorithm

379  ENIGMA_INFO("Successfully auto-detected algorithm used for encryption which is {0}", Algorithm::AlgoTypeEnumToStr(m_type));
380 
381  // Create decryptor based on selected algorithm type
382  const auto algorithm = Algorithm::CreateFromType(m_type, Algorithm::Intent::Decrypt);
383  ENIGMA_ASSERT_OR_THROW(algorithm, ("Failed to create algorithm from type"));
384 
385  // Read in file cipher
386  String cipher{};
387  const bool file_read_success = FileUtils::Read(m_in_filename, cipher);
388  ENIGMA_ASSERT_OR_THROW(file_read_success, ("Failed to read cipher from file {}", m_in_filename));
389  ENIGMA_ASSERT_OR_THROW(!cipher.empty(), ("File {} is empty", m_in_filename));
390 
391  // Decrypt file cipher
392  String buffer = algorithm->Decrypt(m_password, cipher);
393  ENIGMA_ASSERT_OR_THROW(!buffer.empty(), ("Failed to decrypt file cipher"));
394 
395  /*
396  Note: You should compress before encrypting. Encryption turns your data into high-entropy data,
397  usually indistinguishable from a random stream. Compression relies on patterns in order to gain
398  any size reduction. Since encryption destroys such patterns, the compression algorithm would be
399  unable to give you much (if any) reduction in size if you apply it to encrypted data.
400  */
401 
402  // Decompression
403  ENIGMA_TRACE("Decompressing file buffer {0} ...", m_in_filename);
404  auto old_buffer_size = buffer.size();
405  buffer = GZip::Decompress(buffer);
406  auto new_buffer_size = buffer.size();
407  auto increased_bytes = old_buffer_size < new_buffer_size ? (new_buffer_size - old_buffer_size) : 0;
408  ENIGMA_TRACE("File size increased by {0}", SizeUtils::FriendlySize(increased_bytes));
409
