286 void DecryptTextScene::OnDecryptButtonPressed()
287 {
288  if (m_cipher_base64.empty())
289  {
290  (void)DialogUtils::Warn(("Cipher Base64 is empty"));
291  return;
292  }
293  if (m_password.empty())
294  {
295  (void)DialogUtils::Warn(("Encryption password is empty"));
296  return;
297  }
298 
299  try
300  {
301  // Auto detect encryption algorithm
302  m_type = Algorithm::DetectFromCipherBase64(m_cipher_base64);
303  ENIGMA_INFO("Successfully auto-detected algorithm used for encryption which is {0}", Algorithm::AlgoTypeEnumToStr(m_type));
304 
305  // Decode base64 to cipher
306  m_cipher = Base64::Decode(m_cipher_base64);
307  ENIGMA_ASSERT_OR_THROW(!m_cipher.empty(), ("Failed to decode cipher base64! please make sure you have the exact cipher text you received on encryption"));
308 
309  // Create encryptor based on selected algorithm type
310  const auto algorithm = Algorithm::CreateFromType(m_type, Algorithm::Intent::Decrypt);
311  ENIGMA_ASSERT_OR_THROW(algorithm, ("Failed to create algorithm from type"));
312 
313  // Decrypt text
314  m_recovered_text = algorithm->Decrypt(m_password, m_cipher);
315  ENIGMA_ASSERT_OR_THROW(!m_recovered_text.empty(), ("Failed to recover encrypted text"));
316 
317  // Decompress text
318  m_recovered_text = GZip::Decompress(m_recovered_text);
319 
320  // Spawn notification alert if window is not focused
321  //if (!Application::GetInstance()->GetWindow()->IsFocused())
322  //{
323  // Notification{ "Enigma", "Successfully Decrypted Text" }.Show();
324  //}
325  }
326  catch (const CryptoPP::Exception& e)
327  {
328  const String err_msg = CryptoPPUtils::GetFullErrorMessage(e);
329  ENIGMA_ERROR("Decryption Failure: {0}", err_msg);
330  (void)DialogUtils::Error(("Decryption Failure"), err_msg);
331  }
332  catch (const std::exception& e)
333  {
334  ENIGMA_ERROR("Decryption Failure: {0}", e.what());
335  (void)DialogUtils::Error(("Decryption Failure"), e.what());
336  }
337  catch (...)
338  {
339  const String err_msg = ("Decryption Failure UNKNOWN ERRORr");
340  ENIGMA_ERROR("Decryption Failure UNKNOWN ERROR");
341  (void)DialogUtils::Error(err_msg);
342  }
343  }
