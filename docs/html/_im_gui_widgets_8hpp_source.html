93  static bool LoadingSpinner(const char* label, float radius, float thickness, const ImU32 color) {
94  ImGuiWindow* window = ImGui::GetCurrentWindow();
95  if (window->SkipItems)
96  return false;
97 
98  ImGuiContext& g = *GImGui;
99  const ImGuiStyle& style = g.Style;
100  const ImGuiID id = window->GetID(label);
101 
102  ImVec2 pos = window->DC.CursorPos;
103  ImVec2 size((radius) * 2, (radius + style.FramePadding.y) * 2);
104 
105  const ImRect bb(pos, ImVec2(pos.x + size.x, pos.y + size.y));
106  ImGui::ItemSize(bb, style.FramePadding.y);
107  if (!ImGui::ItemAdd(bb, id))
108  return false;
109 
110  // Render
111  window->DrawList->PathClear();
112 
113  int num_segments = 30;
114  int start = (int)std::abs(sinf((float)g.Time * 1.8f) * (num_segments - 5));
115 
116  const float a_min = IM_PI * 2.0f * ((float)start) / (float)num_segments;
117  const float a_max = IM_PI * 2.0f * ((float)num_segments - 3) / (float)num_segments;
118 
119  const ImVec2 centre = ImVec2(pos.x + radius, pos.y + radius + style.FramePadding.y);
120 
121  for (int i = 0; i < num_segments; i++) {
122  const float a = a_min + ((float)i / (float)num_segments) * (a_max - a_min);
123  window->DrawList->PathLineTo(ImVec2(centre.x + cosf(a + (float)g.Time * 8.0f) * radius,
124  centre.y + sinf(a + (float)g.Time * 8.0f) * radius));
125  }
126 
127  window->DrawList->PathStroke(color, false, thickness);
128 
129  return true;
130 
131  }

134  static bool LoadingBar(const char* label, float value, const ImVec2& size_arg, const ImU32 bg_col, const ImU32 fg_col) {
135  ImGuiWindow* window = ImGui::GetCurrentWindow();
136  if (window->SkipItems)
137  return false;
138 
139  ImGuiContext& g = *GImGui;
140  const ImGuiStyle& style = g.Style;
141  const ImGuiID id = window->GetID(label);
142 
143  ImVec2 pos = window->DC.CursorPos;
144  ImVec2 size = size_arg;
145  size.x -= style.FramePadding.x * 2;
146 
147  const ImRect bb(pos, ImVec2(pos.x + size.x, pos.y + size.y));
148  ImGui::ItemSize(bb, style.FramePadding.y);
149  if (!ImGui::ItemAdd(bb, id))
150  return false;
151 
152  // Render
153  const float circleStart = size.x * 0.7f;
154  const float circleEnd = size.x;
155  const float circleWidth = circleEnd - circleStart;
156 
157  window->DrawList->AddRectFilled(bb.Min, ImVec2(pos.x + circleStart, bb.Max.y), bg_col);
158  window->DrawList->AddRectFilled(bb.Min, ImVec2(pos.x + circleStart * value, bb.Max.y), fg_col);
159 
160  const float t = static_cast<float>(g.Time);
161  const float r = size.y / 2;
162  const float speed = 1.5f;
163 
164  const float a = speed * 0;
165  const float b = speed * 0.333f;
166  const float c = speed * 0.666f;
167 
168  const float o1 = (circleWidth + r) * (t + a - speed * (int)((t + a) / speed)) / speed;
169  const float o2 = (circleWidth + r) * (t + b - speed * (int)((t + b) / speed)) / speed;
170  const float o3 = (circleWidth + r) * (t + c - speed * (int)((t + c) / speed)) / speed;
171 
172  window->DrawList->AddCircleFilled(ImVec2(pos.x + circleEnd - o1, bb.Min.y + r), r, bg_col);
173  window->DrawList->AddCircleFilled(ImVec2(pos.x + circleEnd - o2, bb.Min.y + r), r, bg_col);
174  window->DrawList->AddCircleFilled(ImVec2(pos.x + circleEnd - o3, bb.Min.y + r), r, bg_col);
175 
176  return true;
177  }

216  class Image

218  public:
219  explicit Image(const char* file_name)
220  {
221  // Load from file
222  byte* buffer = stbi_load(file_name, &m_width, &m_height, nullptr, 4);
223  ENIGMA_ASSERT(buffer, String("Failed to read image from ") + file_name);
224 
225  // Create a OpenGL texture identifier
226  glAssert( glGenTextures(1, &m_id) );
227  glAssert( glBindTexture(GL_TEXTURE_2D, m_id) );
228 
229  // Setup filtering parameters for display
230  glAssert( glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR) );
231  glAssert( glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR) );
232  glAssert( glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE) ); // This is required on WebGL for non power-of-two textures
233  glAssert( glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE) ); // Same
234 
235  // Upload pixels into texture
236 #if defined(GL_UNPACK_ROW_LENGTH) && !defined(__EMSCRIPTEN__)
237  glAssert( glPixelStorei(GL_UNPACK_ROW_LENGTH, 0) );
238 #endif
239  ENIGMA_INFO("Storing texture #{0} {1} to GPU Memory", m_id, SizeUtils::FriendlySize((size_t)m_width * (size_t)m_height * sizeof(byte)));
240 
241  glAssert( glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, m_width, m_height, 0, GL_RGBA, GL_UNSIGNED_BYTE, buffer) );
242 
243  // at this point the buffer is uploaded to the gpu, no need to keep in the memory
244  stbi_image_free(buffer);
245  }
246 
247  ~Image()
248  {
249  ENIGMA_INFO("Deleting texture #{0} from GPU Memory", m_id);
250  glAssert( glDeleteTextures(1, &m_id) );
251  }
