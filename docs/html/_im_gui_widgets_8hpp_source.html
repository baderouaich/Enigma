309  {
310  // Load from file
311  byte* buffer = stbi_load(file_name, &m_width, &m_height, nullptr, 4);
312  ENIGMA_ASSERT(buffer, String("Failed to read image from ") + file_name);
313 
314  // Create a OpenGL texture identifier
315  glAssert( glGenTextures(1, &m_id) );
316  glAssert( glBindTexture(GL_TEXTURE_2D, m_id) );
317 
318  // Setup filtering parameters for display
319  glAssert( glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR) );
320  glAssert( glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR) );
321  glAssert( glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE) ); // This is required on WebGL for non power-of-two textures
322  glAssert( glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE) ); // Same
323 
324  // Upload pixels into texture
325 #if defined(GL_UNPACK_ROW_LENGTH) && !defined(__EMSCRIPTEN__)
326  glAssert( glPixelStorei(GL_UNPACK_ROW_LENGTH, 0) );
327 #endif
328  ENIGMA_INFO("Storing texture #{0} {1} to GPU Memory", m_id, SizeUtils::FriendlySize((size_t)m_width * (size_t)m_height * sizeof(byte)));
329 
330  glAssert( glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, m_width, m_height, 0, GL_RGBA, GL_UNSIGNED_BYTE, buffer) );
331 
332  // at this point the buffer is uploaded to the gpu, no need to keep in the memory
333  stbi_image_free(buffer);
334  }
